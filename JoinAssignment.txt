-- Join name with film
SELECT WordsByCharacter.character_id,
  character.character,
  character.film,
  character.race,
  character.chapter,
  character.words,
  name.character,
FROM WordsByCharacter
INNER JOIN Movies ON
WordsByCharacter.id = Movies.character_id;


-- Join film with name
SELECT Movies.character_id,
  name.film,
  character.film,
  name.name,
  name.chapter,
  name.race,
  name.words,
FROM Movies
INNER JOIN characters ON
name.id = film.character_id;
